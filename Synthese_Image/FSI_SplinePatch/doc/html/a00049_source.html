00056 inline float sqlength ( const glm::vec3 & v ) {

00061 inline glm::vec3 perpendicular (const glm::vec3 & v) {
00062     static const float fSquareZero = 1e-06 * 1e-06;
00063     glm::vec3 perp = glm::cross(v, glm::vec3(1,0,0));
00064     if (sqlength(perp)< fSquareZero) {
00065             /* This vector is the Y axis multiplied by a scalar, so we have
00066                to use another axis.
00067             */
00068         perp = glm::cross(v, glm::vec3(0,1,0));
00069     }
00070     return perp;
00071 }
